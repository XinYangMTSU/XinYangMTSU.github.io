React + PHP + MySQL JWT Authentication Project

The React + PHP + MySQL JWT Authentication project is a full-stack user authentication system that enables secure user registration, login, and session management using JWT (JSON Web Token). The frontend, built with React, provides a seamless user interface for the account creation and authentication, sending API requests to a PHP-based backend. The backend, powered by PHP and MySQL, handles user data storage, password hashing for security, and JWT generation for authentication. Users register with a username and password, which are securely stored in a MySQL database. Upon login, a JWT token is issued, allowing authenticated users to access protected routes. The system ensures security with password hashing (bcrypt) and JWT-based session management, making it a robust and scalable authentication solution.

D. Create JWT Helper

  • ★ Create jwt_helper.php, this file generates & verifies JWT tokens.
			
				<?php
				require __DIR__ . "/vendor/autoload.php";
				use Firebase\JWT\JWT;
				use Firebase\JWT\Key;
				
				$secret_key = "your_secret_key_here";
				
				function createJWT($user_id, $username) {
				    global $secret_key;
				    $payload = [
				        "id" => $user_id,
				        "username" => $username,
				        "exp" => time() + (60 * 60) // Token expires in 1 hour
				    ];
				    return JWT::encode($payload, $secret_key, "HS256");
				}
				
				function verifyJWT($token) {
				    global $secret_key;
				    try {
				        return JWT::decode($token, new Key($secret_key, "HS256"));
				    } catch (Exception $e) {
				        return false;
				    }
				}
				?>

E. Create User Registration API

  • ★ Create register.php, handles user signup and password hashing.
			
			<?php
				header("Access-Control-Allow-Origin: *");
				header("Content-Type: application/json");
				include "db.php";
				
				$data = json_decode(file_get_contents("php://input"));
				
				if (!empty($data->username) && !empty($data->password)) {
				    $username = $data->username;
				    $password = password_hash($data->password, PASSWORD_DEFAULT); // Hash password
				
				    $sql = "INSERT INTO users (username, password) VALUES (?, ?)";
				    $stmt = $conn->prepare($sql);
				    $stmt->bind_param("ss", $username, $password);
				
				    if ($stmt->execute()) {
				        echo json_encode(["message" => "User registered successfully!"]);
				    } else {
				        echo json_encode(["error" => "User registration failed."]);
				    }
				} else {
				    echo json_encode(["error" => "All fields are required."]);
				}
			?>

F. Create Login API

  • ★ Create login.php, handles user login & JWT token generation.
			
			<?php
				header("Access-Control-Allow-Origin: *");
				header("Content-Type: application/json");
				include "db.php";
				include "jwt_helper.php";
				
				$data = json_decode(file_get_contents("php://input"));
				
				if (!empty($data->username) && !empty($data->password)) {
				    $username = $data->username;
				    $password = $data->password;
				
				    $sql = "SELECT * FROM users WHERE username = ?";
				    $stmt = $conn->prepare($sql);
				    $stmt->bind_param("s", $username);
				    $stmt->execute();
				    $result = $stmt->get_result();
				
				    if ($result->num_rows > 0) {
				        $user = $result->fetch_assoc();
				        if (password_verify($password, $user["password"])) {
				            $token = createJWT($user["id"], $user["username"]);
				            echo json_encode(["message" => "Login successful!", "token" => $token]);
				        } else {
				            echo json_encode(["error" => "Invalid password."]);
				        }
				    } else {
				        echo json_encode(["error" => "User not found."]);
				    }
				} else {
				    echo json_encode(["error" => "Username and password are required."]);
				}
			?>

3. Set Up React Frontend

A. Create a React App

			npx create-react-app react-auth
			cd react-auth

Install Axios

			npm install axios

C. Implement Authentication in React

  • ★ Modify App.js:
			
		import React, { useState } from "react";
		import axios from "axios";
		
		function App() {
		    const [token, setToken] = useState(null);
		    const [username, setUsername] = useState("");
		    const [password, setPassword] = useState("");
		
		    const handleRegister = async () => {
			const response = await axios.post("http://localhost/react-php-auth/register.php", 
			    { username, password },
			    { headers: { "Content-Type": "application/json" } }
			);
			alert(response.data.message || response.data.error);
		    };
		
		    const handleLogin = async () => {
			const response = await axios.post("http://localhost/react-php-auth/login.php", 
			    { username, password },
			    { headers: { "Content-Type": "application/json" } }
			);
			if (response.data.token) {
			    setToken(response.data.token);
			    localStorage.setItem("token", response.data.token);
			    alert("Login successful!");
			} else {
			    alert(response.data.error);
			}
		    };
		
		    const handleLogout = () => {
			setToken(null);
			localStorage.removeItem("token");
		    };
		
		    return (
			<div>
			    <h1>React + PHP + MySQL JWT Auth</h1>
			    {token ? (
				<>
				    <p>Logged in! Token: {token}</p>
				    <button onClick={handleLogout}>Logout</button>
				</>
			    ) : (
				<>
				    <input type="text" placeholder="Username" onChange={(e) => setUsername(e.target.value)} />
				    <input type="password" placeholder="Password" onChange={(e) => setPassword(e.target.value)} />
				    <button onClick={handleRegister}>Register</button>
				    <button onClick={handleLogin}>Login</button>
				</>
			    )}
			</div>
		    );
		}
		
		export default App;
