Removing Duplicates in Python

  
    # Imports the pandas library and gives it the shorthand name "pd"
    import pandas as pd
    
    # Create simple dataset with duplicates
    
    # Creates a dictionary with two lists: names and ages
    data = {'Name': ['Alice', 'Bob', 'Alice', 'Charlie'], 
            'Age': [25, 30, 25, 35]}
    
    # pd.DataFrame(data) converts the dictionary into a pandas DataFrame (like a table)
    df = pd.DataFrame(data)
    
    print("Original data before dropping duplicates:")
    print(df)
    print(f"\nRows before dropping duplicates: {len(df)}")
    
    # Check if ANY duplicates exist
    has_duplicates = df.duplicated().any()
    print(f"\nHas duplicates: {has_duplicates}")  # True
    
    # Remove duplicates
    df_clean = df.drop_duplicates()
    print(f"\nRows after dropping duplicates: {len(df_clean)}")
    print("\nCleaned data after dropping duplicates::")
    print(df_clean)
